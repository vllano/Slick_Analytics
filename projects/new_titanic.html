# Missing data per column
def null_data(x):
    total = x.isnull().sum().sort_values(ascending=False)
    percent = (x.isnull().sum()/x.isnull().count()).sort_values(ascending=False)
    missing_data = pd.concat([total, percent], axis=1, keys=['Total', 'Percent'])
    return missing_data

# Function for filling the nulls of'Age'with median by 'Pclass'
def age_class(x):
    fstclsavg = x[x['Pclass'] == 1]
    fsttotal = fstclsavg['Pclass'].count()        
    scdclsavg = x[x['Pclass'] == 2]
    scdtotal = scdclsavg['Pclass'].count()
    trdclsavg = x[x['Pclass'] == 3]
    trdtotal = trdclsavg['Pclass'].count()
    age1st = fstclsavg['Age'].median()
    age2nd = scdclsavg['Age'].median()
    age3rd = trdclsavg['Age'].median()
    x['Age'].loc[x['Pclass'] == 1] = x['Age'].fillna(age1st)
    x['Age'].loc[x['Pclass'] == 2] = x['Age'].fillna(age2nd)
    x['Age'].loc[x['Pclass'] == 3] = x['Age'].fillna(age3rd)
    print('1st class total:', fsttotal, '| Age median:', age1st)
    print('2nd class total:', scdtotal,'| Age median:', age2nd)
    print('3rd class total:', trdtotal,'| Age median:', age3rd, '\n')

def plot_confusion_matrix(cm, classes,
                          normalize=False,
                          title='Confusion matrix',
                          cmap=plt.cm.Blues):
    """
    This function prints and plots the confusion matrix.
    Normalization can be applied by setting `normalize=True`.
    """
    if normalize:
        cm = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis]
        print("Normalized confusion matrix")
    else:
        print('Confusion matrix, without normalization')

    print(cm)

    plt.imshow(cm, interpolation='nearest', cmap=cmap)
    plt.title(title)
    plt.colorbar()
    tick_marks = np.arange(len(classes))
    plt.xticks(tick_marks, classes, rotation=45)
    plt.yticks(tick_marks, classes)

    fmt = '.2f' if normalize else 'd'
    thresh = cm.max() / 2.
    for i, j in itertools.product(range(cm.shape[0]), range(cm.shape[1])):
        plt.text(j, i, format(cm[i, j], fmt),
                 horizontalalignment="center",
                 color="white" if cm[i, j] > thresh else "black")

    plt.ylabel('True label')
    plt.xlabel('Predicted label')
    plt.tight_layout()
