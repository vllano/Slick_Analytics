import pandas as pd
import numpy as np
from datetime import datetime, timedelta
import matplotlib.pyplot as plt
import seaborn as sns
from scipy.stats import ttest_ind, chi2_contingency

def generate_ab_testing_data(n_users=10000):
    """
    Generates synthetic data for A/B testing analysis with statistically significant results.
    
    Args:
        n_users: The number of users to simulate.
    
    Returns:
        A Pandas DataFrame containing the synthetic A/B testing data.
    """
    np.random.seed(42)
    
    # User demographic data
    user_ids = np.arange(1, n_users + 1)
    age = np.random.randint(18, 65, n_users)
    gender = np.random.choice(['Male', 'Female', 'Other'], n_users)
    location = np.random.choice(['Urban', 'Suburban', 'Rural'], n_users)
    device = np.random.choice(['Desktop', 'Mobile', 'Tablet'], n_users)

    # Simulate interaction dates over a period (e.g., last 90 days)
    start_date = datetime.today() - timedelta(days=30)
    dates = [start_date + timedelta(days=np.random.randint(0, 30)) for _ in range(n_users)]
    dates = pd.to_datetime(dates).date 
    
    # A/B testing data
    group = np.random.choice(['A', 'B'], n_users, p=[0.5, 0.5])
    page_views = np.random.randint(1, 10, n_users)
    time_on_page_a = np.random.exponential(scale=60, size=n_users)
    time_on_page_b = np.random.exponential(scale=75, size=n_users)
    time_on_page = np.where(group == 'A', time_on_page_a, time_on_page_b)
    time_on_page = np.round(time_on_page, 2)
    
    # Lead funnel stage data 
    # Step 1: All users are aware
    awareness = np.full(n_users, True)
    
    # Step 2: Interest only if aware
    raw_interest = np.where(group == 'A',
                            np.random.choice([True, False], n_users, p=[0.50, 0.50]),
                            np.random.choice([True, False], n_users, p=[0.70, 0.30]))
    interest = raw_interest & awareness  # Only interested if aware
    
    # Step 3: Consideration only if interested
    raw_consideration = np.where(group == 'A',
                                 np.random.choice([True, False], n_users, p=[0.35, 0.65]),
                                 np.random.choice([True, False], n_users, p=[0.50, 0.50]))
    consideration = raw_consideration & interest  # Only consider if interested
    
    # Step 4: Decision only if considered
    raw_decision = np.where(group == 'A',
                            np.random.choice([True, False], n_users, p=[0.15, 0.85]),
                            np.random.choice([True, False], n_users, p=[0.30, 0.70]))
    decision = raw_decision & consideration  # Only decide if considered
    
        
    # Simulate conversions with different probabilities for each group,
    # ensuring conversions are not more than decisions
    def generate_conversions(group_val, decision_val):
        if not decision_val:
            return False
        if group_val == 'A':
            return np.random.choice([True, False], p=[0.6, 0.4]) #Max 60% of decisions convert
        else:
            return np.random.choice([True, False], p=[0.70, 0.30]) #Max 80% of decisions convert
    
    conversion = np.array([generate_conversions(g, d) for g, d in zip(group, decision)])
    
    # Simulate revenue per conversion
    avg_revenue_a = 100
    avg_revenue_b = 120
    revenue_a = np.where(conversion & (group == 'A'), np.random.normal(loc=avg_revenue_a, scale=30, size=n_users).clip(0), 0)
    revenue_b = np.where(conversion & (group == 'B'), np.random.normal(loc=avg_revenue_b, scale=40, size=n_users).clip(0), 0)
    revenue = np.where(group == 'A', revenue_a, revenue_b)
    revenue = np.round(revenue, 2)
    
    # Combine data into a dictionary
    data = {
        'user_id': user_ids,
        'date': dates, 
        'age': age,
        'gender': gender,
        'location': location,
        'device': device,
        'group': group,
        'page_views': page_views,
        'time_on_page': time_on_page,
        'conversion': conversion,
        'revenue': revenue,
        'awareness': awareness,
        'interest': interest,
        'consideration': consideration,
        'decision': decision,
    }
    
    # Create a Pandas DataFrame
    df = pd.DataFrame(data)
    return df
    
df = generate_ab_testing_data()

# 12. Statistical Significance of Conversion Rate Difference
print("\n12. Statistical Significance of Conversion Rate Difference:\n")
group_a_conversions = df[df['group'] == 'A']['conversion']
group_b_conversions = df[df['group'] == 'B']['conversion']
t_statistic, p_value = ttest_ind(group_a_conversions, group_b_conversions)
print(f"T-statistic: {t_statistic:.3f}, P-value: {p_value:.3f}")
if p_value < 0.05:
    print("The difference in conversion rates between Group A and Group B is statistically significant.")
else:
    print("The difference in conversion rates between Group A and Group B is not statistically significant.")

# 13. Chi-Square Test for Funnel Stage Progression
print("\n13. Chi-Square Test for Funnel Stage Progression:")
for stage in ['interest', 'consideration', 'decision', 'conversion']:
    print(f"\nChi-Square Test for {stage.capitalize()} vs. Group:")
    contingency_table = pd.crosstab(df['group'], df[stage])
    chi2, p, dof, expected = chi2_contingency(contingency_table)
    print(f"Chi2: {chi2:.3f}, P-value: {p:.3f}, Degrees of Freedom: {dof}")
    if p < 0.05:
        print(f"  There is a statistically significant association between group and {stage}.")
    else:
        print(f"  There is no statistically significant association between group and {stage}.")
